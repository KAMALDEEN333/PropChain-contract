245    /// Dynamic fee calculation: base * (1 + congestion_factor + demand_factor)
246    fn compute_dynamic_fee(
247        config: &FeeConfig,
248        congestion_index: u32,
249        demand_factor_bp: u32,
250    ) -> u128 {
251        // Congestion multiplier: 0-100 -> 0% to (MAX_CONGESTION_MULTIPLIER-100)%
252        let congestion_bp = (congestion_index as u128)
253            .saturating_mul(config.congestion_sensitivity as u128)
254            .saturating_mul((MAX_CONGESTION_MULTIPLIER - 100) as u128)
255            / 10_000;
256        let demand_bp = demand_factor_bp.min(5000); // Cap demand at 50%
257        let total_multiplier_bp = 10_000u128
258            .saturating_add(congestion_bp)
259            .saturating_add(demand_bp as u128);
260        let fee = config
261            .base_fee
262            .saturating_mul(total_multiplier_bp)
263            .saturating_div(BASIS_POINTS);
264        fee.clamp(config.min_fee, config.max_fee)
265    }

316        /// Compute current congestion index (0-100) from recent activity
317        fn congestion_index(&self) -> u32 {
318            let now = self.env().block_timestamp();
319            let window_secs = 3600u64; // 1 hour window
320            if now.saturating_sub(self.last_congestion_reset) > window_secs {
321                return 0; // Reset after window
322            }
323            let count = self.recent_ops_count;
324            // Normalize to 0-100: CONGESTION_WINDOW ops = 100
325            (count.saturating_mul(100).saturating_div(CONGESTION_WINDOW)).min(100)
326        }
327
328        /// Demand factor in basis points (from recent volume)
329        fn demand_factor_bp(&self) -> u32 {
330            let ci = self.congestion_index();
331            self.default_config
332                .demand_factor_bp
333                .saturating_mul(ci)
334                .saturating_div(100)
335        }

367            self.total_fees_collected = self.total_fees_collected.saturating_add(amount);
368            Ok(())

373        /// Automated fee adjustment based on recent utilization vs target
374        #[ink(message)]
375        pub fn update_fee_params(&mut self) -> Result<(), FeeError> {
376            self.ensure_admin()?;
377            let now = self.env().block_timestamp();
378            let congestion = self.congestion_index();
379            let mut config = self.default_config.clone();
380            if congestion > 70 {
381                config.base_fee = config
382                    .base_fee
383                    .saturating_mul(105)
384                    .saturating_div(100)
385                    .min(config.max_fee);
386            } else if congestion < 30 {
387                config.base_fee = config
388                    .base_fee
389                    .saturating_mul(95)
390                    .saturating_div(100)
391                    .max(config.min_fee);
392            }
393            config.last_updated = now;
394            self.default_config = config.clone();
395            self.env().emit_event(FeeConfigUpdated {
396                by: self.env().caller(),
397                operation: None,
398                base_fee: config.base_fee,
399                timestamp: now,
400            });
401            Ok(())
402        }

427        /// Create premium listing auction (pay fee; fee goes to treasury)
428        #[ink(message)]
429        pub fn create_premium_auction(
430            &mut self,
431            property_id: u64,
432            min_bid: u128,
433            duration_seconds: u64,
434        ) -> Result<u64, FeeError> {
435            let caller = self.env().caller();
436            let now = self.env().block_timestamp();
437            let fee = self.calculate_fee(FeeOperation::PremiumListingBid);
438            if fee > 0 {
439                self.fee_treasury = self.fee_treasury.saturating_add(fee);
440                self.total_fees_collected = self.total_fees_collected.saturating_add(fee);
441            }
442            self.auction_count += 1;
443            let auction_id = self.auction_count;
444            let auction = PremiumAuction {
445                property_id,
446                seller: caller,
447                min_bid,
448                current_bid: 0,
449                current_bidder: None,
450                end_time: now.saturating_add(duration_seconds),
451                settled: false,
452                fee_paid: fee,
453            };
454            self.auctions.insert(auction_id, &auction);
455            self.env().emit_event(PremiumAuctionCreated {
456                auction_id,
457                property_id,
458                seller: caller,
459                min_bid,
460                end_time: auction.end_time,
461                fee_paid: fee,
462            });
463            Ok(auction_id)
464        }

466        /// Place or increase bid (bid must be > current_bid and >= min_bid)
467        #[ink(message)]
468        pub fn place_bid(&mut self, auction_id: u64, amount: u128) -> Result<(), FeeError> {
469            let caller = self.env().caller();
470            let now = self.env().block_timestamp();
471            let mut auction = self
472                .auctions
473                .get(auction_id)
474                .ok_or(FeeError::AuctionNotFound)?;
475            if auction.settled {
476                return Err(FeeError::AlreadySettled);
477            }
478            if now >= auction.end_time {
479                return Err(FeeError::AuctionEnded);
480            }
481            if amount < auction.min_bid {
482                return Err(FeeError::BidTooLow);
483            }
484            if amount <= auction.current_bid {
485                return Err(FeeError::BidTooLow);
486            }
487            let outbid = auction.current_bid;
488            auction.current_bid = amount;
489            auction.current_bidder = Some(caller);
490            self.auctions.insert(auction_id, &auction);
491            self.auction_bids.insert(
492                (auction_id, caller),
493                &AuctionBid {
494                    bidder: caller,
495                    amount,
496                    timestamp: now,
497                },
498            );
499            self.env().emit_event(PremiumAuctionBid {
500                auction_id,
501                bidder: caller,
502                amount,
503                outbid_previous: outbid,
504            });
505            Ok(())
506        }

508        /// Settle auction after end_time; winner is current_bidder
509        #[ink(message)]
510        pub fn settle_auction(&mut self, auction_id: u64) -> Result<(), FeeError> {
511            let now = self.env().block_timestamp();
512            let mut auction = self
513                .auctions
514                .get(auction_id)
515                .ok_or(FeeError::AuctionNotFound)?;
516            if auction.settled {
517                return Err(FeeError::AlreadySettled);
518            }
519            if now < auction.end_time {
520                return Err(FeeError::AuctionNotEnded);
521            }
522            let winner = auction.current_bidder.ok_or(FeeError::AuctionNotFound)?;
523            let amount = auction.current_bid;
524            auction.settled = true;
525            self.auctions.insert(auction_id, &auction);
526            // fee_paid was already added to fee_treasury at auction creation
527            self.env().emit_event(PremiumAuctionSettled {
528                auction_id,
529                property_id: auction.property_id,
530                winner,
531                amount,
532                timestamp: now,
533            });
534            Ok(())
535        }

583        /// Distribute accumulated fees: validator share to validators, rest to treasury
584        #[ink(message)]
585        pub fn distribute_fees(&mut self) -> Result<(), FeeError> {
586            self.ensure_admin()?;
587            let amount = self.fee_treasury;
588            if amount == 0 {
589                return Ok(());
590            }
591            let validator_total = amount
592                .saturating_mul(self.validator_share_bp as u128)
593                .saturating_div(BASIS_POINTS);
594            let validator_list = self.validator_list.clone();
595            let validator_count = validator_list.len() as u32;
596            if validator_count > 0 && validator_total > 0 {
597                let per_validator = validator_total.saturating_div(validator_count as u128);
598                for acc in validator_list {
599                    let current = self.pending_rewards.get(acc).unwrap_or(0);
600                    self.pending_rewards
601                        .insert(acc, &current.saturating_add(per_validator));
602                    self.record_reward(acc, per_validator, RewardReason::ValidatorReward);
603                    self.total_distributed = self.total_distributed.saturating_add(per_validator);
604                    self.env().emit_event(RewardsDistributed {
605                        recipient: acc,
606                        amount: per_validator,
607                        reason: RewardReason::ValidatorReward,
608                        timestamp: self.env().block_timestamp(),
609                    });
610                }
611            }
612            self.fee_treasury = 0;
613            Ok(())
614        }

660        /// Fee estimate with optimization recommendation
661        #[ink(message)]
662        pub fn get_fee_estimate(&self, operation: FeeOperation) -> FeeEstimate {
663            let config = self.get_config(operation);
664            let congestion = self.congestion_index();
665            let demand_bp = self.demand_factor_bp();
666            let estimated = compute_dynamic_fee(&config, congestion, demand_bp);
667            let congestion_level = if congestion < 33 {
668                "low"
669            } else if congestion < 66 {
670                "medium"
671            } else {
672                "high"
673            };
674            let recommendation = if congestion >= 70 {
675                "Consider batching operations or submitting during off-peak."
676            } else if congestion < 30 {
677                "Good time to submit; fees are below average."
678            } else {
679                "Fees are at typical levels."
680            };
681            FeeEstimate {
682                operation,
683                estimated_fee: estimated,
684                min_fee: config.min_fee,
685                max_fee: config.max_fee,
686                congestion_level: congestion_level.into(),
687                recommendation: recommendation.into(),
688            }
689        }

691        /// Full fee report for transparency and dashboard
692        #[ink(message)]
693        pub fn get_fee_report(&self) -> FeeReport {
694            let now = self.env().block_timestamp();
695            let recommended = self.calculate_fee(FeeOperation::RegisterProperty);
696            let mut active_auctions = 0u32;
697            for id in 1..=self.auction_count {
698                if let Some(a) = self.auctions.get(id) {
699                    if !a.settled && now < a.end_time {
700                        active_auctions += 1;
701                    }
702                }
703            }
704            FeeReport {
705                config: self.default_config.clone(),
706                congestion_index: self.congestion_index(),
707                recommended_fee: recommended,
708                total_fees_collected: self.total_fees_collected,
709                total_distributed: self.total_distributed,
710                operation_count_24h: self.recent_ops_count as u64,
711                premium_auctions_active: active_auctions,
712                timestamp: now,
713            }
714        }

717        #[ink(message)]
718        pub fn get_fee_recommendations(&self) -> Vec<String> {
719            let mut rec = Vec::new();
720            let c = self.congestion_index();
721            if c >= 70 {
722                rec.push("High congestion: use batch operations to reduce total fee.".into());
723                rec.push("Consider submitting during off-peak hours.".into());
724            } else if c < 30 {
725                rec.push("Low congestion: current fees are favorable.".into());
726            }
727            rec.push("Premium listings: use auctions for better price discovery.".into());
728            rec.push("Check get_fee_estimate before each operation type.".into());
729            rec
730        }
